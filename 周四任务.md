## 第二步：插件端注册频道并接收包

实现 `PluginMessageListener`，注册频道，按 key 分发处理：

```kotlin
@PlatformSide(Platform.BUKKIT)
object AuctionPlugin : Plugin(), PluginMessageListener {

    @Awake(LifeCycle.ENABLE)
    fun registerChannel() {
        // 注册接收频道（C2S）
        Bukkit.getServer().messenger.registerIncomingPluginChannel(
            BukkitPlugin.getInstance(), "legendengine:auction", this
        )
        // 注册发送频道（S2C，必须注册才能 sendPluginMessage）
        Bukkit.getServer().messenger.registerOutgoingPluginChannel(
            BukkitPlugin.getInstance(), "legendengine:data_packet"
        )
    }

    override fun onPluginMessageReceived(channel: String, player: Player, message: ByteArray) {
        val stream = DataInputStream(ByteArrayInputStream(message))
        when (channel) {
            "legendengine:auction" -> {
                val key   = readMcString(stream)  // "list"
                val value = readMcString(stream)  // 页码
                when (key) {
                    "list" -> AuctionManager.queryPage(player, value.toIntOrNull() ?: 1)
                }
            }
        }
    }

    // 解包工具：VarInt 前缀 UTF-8，与 FriendlyByteBuf.writeUtf 格式一致
    private fun readVarInt(stream: DataInputStream): Int {
        var result = 0; var shift = 0; var b: Int
        do { b = stream.readByte().toInt() and 0xFF
             result = result or ((b and 0x7F) shl shift); shift += 7
        } while ((b and 0x80) != 0)
        return result
    }
    private fun readMcString(stream: DataInputStream): String {
        val bytes = ByteArray(readVarInt(stream))
        stream.readFully(bytes)
        return String(bytes, Charsets.UTF_8)
    }
}
```

## 第三步：服务端查数据并回包

### 数据层架构：Caffeine + Redis + MySQL(可以考虑 MongoDB，据说比redis好用点)

```
请求到达
   ↓
Caffeine（JVM 本地缓存，毫秒级）
   ↓ 未命中
Redis（分布式缓存，毫秒级，设 TTL）
   ↓ 未命中
MySQL（权威数据来源，异步查询）
   ↓
写回 Redis → 写回 Caffeine → 响应客户端
```

| 层 | 框架 | 作用 | TTL |
|----|------|------|-----|
| L1 | Caffeine | JVM 进程内热点页缓存，避免打穿 Redis | 30 秒 |
| L2 | Redis | 跨进程共享缓存，多服共用同一份数据 | 5 分钟 |
| 持久化 | MySQL | 权威数据，上架/购买/下架都写这里 | — |

**异步读取 + 乐观锁**

```kotlin
object AuctionManager {

    private val localCache: Cache<String, String> = Caffeine.newBuilder()
        .expireAfterWrite(30, TimeUnit.SECONDS)
        .maximumSize(200)
        .build()

    fun queryPage(player: Player, page: Int) {
        val cacheKey = "auction:page:$page"

        // 1. 查 Caffeine
        val local = localCache.getIfPresent(cacheKey)
        if (local != null) { sendToClient(player, local); return }

        // 2. 异步查 Redis → MySQL
        CompletableFuture.supplyAsync {
            val redis = BackpackPlugin.redisClient.sync()
            val cached = redis.get(cacheKey)
            if (cached != null) {
                localCache.put(cacheKey, cached)
                return@supplyAsync cached
            }
            // 3. 查 MySQL，乐观锁：UPDATE ... WHERE id=? AND version=?
            val items = Database.queryAuctionPage(page)
            val json  = Gson().toJson(items)
            redis.setex(cacheKey, 300, json)  // TTL 5分钟
            localCache.put(cacheKey, json)
            json
        }.thenAccept { json ->
            Bukkit.getScheduler().runTask(BukkitPlugin.getInstance(), Runnable {
                sendToClient(player, json)
            })
        }
    }
}
```

**写操作（上架/购买）流程**
1. MySQL 写入，带 `version` 乐观锁，冲突则重试或返回失败提示
2. 写成功后 `DEL auction:page:*`，让下次查询重新拉库
3. Caffeine 同步失效

**Redis 内存管理**
- `maxmemory-policy allkeys-lru`：超限自动淘汰最久未用的键
- 定时任务每小时主动扫描并删除已过期拍卖条目，防止 key 堆积

## 第四步：Redis 条目数据结构

拍卖行条目存于 Redis，key 格式：`auction:{uuid}`

```json
{
  "uuid":        "1234567890123456789",
  "name":        "diamond_sword", 
  "amount":      1,
  "price":       5000,
  "seller_uuid": "xxxx-xxxx-xxxx-xxxx",
  "expire_at":   1888
}
```

| 字段 | 说明                                         |
|------|--------------------------------------------|
| `uuid` | uuid随机算法生成，唯一标识此条拍卖                        |
| `name` | 物品复合标识，冒号分隔：模板ID : 强化等级 : lore等级 : 自定义模型数据 |
| `amount` | 数量，上架界面玩家手动输入                              |
| `price` | 价格，上架界面玩家手动输入                              |
| `seller_uuid` | 卖家 UUID，系统自动填入                             |
| `expire_at` | 过期时间，上架时玩家自定义输入（单位：小时）                     |

---
格式就是把当页所有条目打成一个数组：                                                                                                                                                                                  
{                                                                                                                                                                                                                  ─────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
"page": 1,
"total": 100,
"items": [
{ "uuid": "XXX", "name": "diamond_sword", "amount": 1, "price": 5000, "seller_uuid": "xxx", "expire_at": 1888 },
{ "uuid": "XXX", "name": "iron_sword", "amount": 2, "price": 1000, "seller_uuid": "yyy", "expire_at": 1888 }                                                                                                       ]

## 第五步：客户端收包动态刷新 UI

服务端回包后，客户端用 `Plugin.listen` 注册监听器接收，在 `onOpen` 时注册，收到数据直接刷新列表：

```lua
-- Script/UI/auction.lua
function onOpen()
    -- 注册监听，收到包就刷新
    Plugin.listenJson("legendengine:auction", function(data)
        refreshList(data.items)
        updatePageBar(data.page, data.total)
    end)
    -- 请求第一页
    Plugin.send("legendengine:auction", "list", "1")
end

function onClose()
    Plugin.clearListeners("legendengine:auction")
end

function refreshList(items)
    local list = ui:findWidget("item_list")
    list:clearChildren()
    for _, item in ipairs(items) do
        local row = list:addRow()
        row:setText("name",   item.name)
        row:setText("price",  item.price .. " 金")
        row:setText("seller", item.seller)
        row:setOnClick(function() onSelectItem(item) end)
    end
end

function updatePageBar(page, total)
    ui:findWidget("page_label"):setText(page .. " / " .. math.ceil(total / 20))
    ui:findWidget("btn_prev"):setEnabled(page > 1)
    ui:findWidget("btn_next"):setEnabled(page * 20 < total)
end
```
